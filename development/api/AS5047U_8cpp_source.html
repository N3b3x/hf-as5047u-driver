129template <typename SpiType>
130uint16_t AS5047U<SpiType>::GetZeroPosition(uint8_t retries) const {
131 uint8_t m = 0;
132 uint8_t l = 0;
133 constexpr uint16_t retryMask = static_cast<uint16_t>(AS5047U_Error::CrcError) |
134 static_cast<uint16_t>(AS5047U_Error::FramingError);
135
136 // First read ZPOSM with retries
137 for (uint8_t i = 0; i <= retries; ++i) {
138 m = this->template ReadReg<AS5047U_REG::ZPOSM>().bits.ZPOSM_bits;
139 auto err = GetStickyErrorFlags();
140 if (!(static_cast<uint16_t>(err) & retryMask))
141 break;
142 }
143
144 // Then read ZPOSL with retries
145 for (uint8_t i = 0; i <= retries; ++i) {
146 l = this->template ReadReg<AS5047U_REG::ZPOSL>().bits.ZPOSL_bits;
147 auto err = GetStickyErrorFlags();
148 if (!(static_cast<uint16_t>(err) & retryMask))
149 break;
150 }
151
152 return static_cast<uint16_t>((m << 6) | l);
153}

193// | 0 | 1 | 1 | PWM | W |
194// | 1 | 1 | x | I | W |
195// | 0 | 0 | 1 | - | PWM |
196// | 0 | 0 | 0 | - | - |

199bool AS5047U<SpiType>::ConfigureInterface(bool abi, bool uvw, bool pwm, uint8_t retries) {
200 auto dis = this->template ReadReg<AS5047U_REG::DISABLE>();
201 auto s2 = this->template ReadReg<AS5047U_REG::SETTINGS2>();
202 dis.bits.ABI_off = abi ? 0 : 1;
203 dis.bits.UVW_off = uvw ? 0 : 1;
204 if (abi && !uvw) {
205 s2.bits.UVW_ABI = 0;
206 s2.bits.PWMon = pwm;
207 } else if (!abi && uvw) {
208 s2.bits.UVW_ABI = 1;
209 s2.bits.PWMon = pwm;
210 } else {
211 s2.bits.UVW_ABI = 0;
212 s2.bits.PWMon = pwm;
213 }
214 return this->template WriteReg(dis, retries) && this->template WriteReg(s2, retries);
215}

248template <typename SpiType>
249bool AS5047U<SpiType>::ProgramOTP() {
250 // Save current frame format and ensure we use CRC for OTP programming
251 FrameFormat backup = this->frame_format_;
252 if (this->frame_format_ == FrameFormat::SPI_16) {
253 this->frame_format_ = FrameFormat::SPI_24;
254 }
255
256 // Set current angle as zero position
257 SetZeroPosition(GetAngle());
258
259 // Backup the volatile shadow registers that will be committed to OTP
260 uint16_t volatile_shadow[5];
261 for (uint16_t a = 0x0016; a <= 0x001A; ++a) {
262 volatile_shadow[a - 0x0016] = readRegister(a);
263 }
264
265 // Enable ECC and compute needed checksum
266 auto ecc = this->template ReadReg<AS5047U_REG::ECC>();
267 ecc.bits.ECC_en = 1;
268 this->template WriteReg(ecc);
269 auto key = this->template ReadReg<AS5047U_REG::ECC_Checksum>().bits.ECC_s;
270 ecc.bits.ECC_chsum = key;
271 this->template WriteReg(ecc);
272
273 // Verify shadow registers are still correct
274 for (uint16_t a = 0x0016; a <= 0x001A; ++a) {
275 if (readRegister(a) != volatile_shadow[a - 0x0016]) {
276 this->frame_format_ = backup;
277 return false;
278 }
279 }
280
281 // Begin OTP programming sequence
282 AS5047U_REG::PROG p{};
283 p.bits.PROGEN = 1;
284 this->template WriteReg(p);
285 p.bits.PROGOTP = 1;
286 this->template WriteReg(p);
287
288 // Wait for programming to complete (timeout after ~15000 cycles)
289 for (uint16_t i = 0; i < 15000; ++i) {
290 if (readRegister(AS5047U_REG::PROG::ADDRESS) == 0x0001) {
291 this->frame_format_ = backup;
292
293 // Guard-band verification: enable PROGVER and refresh OTPREF
294 p.bits.PROGVER = 1;
295 this->template WriteReg(p);
296
297 // Toggle OTPREF to reload from OTP
298 p.bits.OTPREF = 1;
299 this->template WriteReg(p);
300 p.bits.OTPREF = 0;
301 this->template WriteReg(p);
302
303 // Verify shadow registers match OTP
304 for (uint16_t a = 0x0016; a <= 0x001A; ++a) {
305 if (readRegister(a) != volatile_shadow[a - 0x0016]) {
306 this->frame_format_ = backup;
307 return false;
308 }
309 }
310 return true;
311 }
312 }
313
314 // Restore original frame format and return failure if timeout
315 this->frame_format_ = backup;
316 return false;
317}

319template <typename SpiType>
320void AS5047U<SpiType>::updateStickyErrors(uint16_t err_fl) const {
321 // Map ERRFL bits (0-10) to sticky error enum
322 if (err_fl & (1u << 0))
323 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::AgcWarning);
324 if (err_fl & (1u << 1))
325 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::MagHalf);
326 if (err_fl & (1u << 2))
327 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::P2ramWarning);
328 if (err_fl & (1u << 3))
329 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::P2ramError);
330 if (err_fl & (1u << 4))
331 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::FramingError);
332 if (err_fl & (1u << 5))
333 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::CommandError);
334 if (err_fl & (1u << 6))
335 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::CrcError);
336 if (err_fl & (1u << 7))
337 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::WatchdogError);
338 if (err_fl & (1u << 9))
339 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::OffCompError);
340 if (err_fl & (1u << 10))
341 sticky_errors_ |= static_cast<uint16_t>(AS5047U_Error::CordicOverflow);
342}

393// Low level register read without sticky error update
394template <typename SpiType>
395uint16_t AS5047U<SpiType>::rawReadRegister(uint16_t address) const {
396 uint16_t result = 0;
397 if (this->frame_format_ == FrameFormat::SPI_16) {
398 // ---- 16-bit frame without CRC ----
399 // Prepare read command (bit14=1 for read)
400 uint16_t cmd = static_cast<uint16_t>(0x4000 | (address & 0x3FFF));
401 const uint8_t tx[2] = {static_cast<uint8_t>(cmd >> 8), static_cast<uint8_t>(cmd & 0xFF)};
402 uint8_t rx[2];
403
404 // Send read command
405 spi_.transfer(tx, rx, 2);
406
407 // Send NOP to receive register data
408 const uint8_t tx_nop[2] = {0x00, 0x00};
409 uint8_t rx_data[2];
410 spi_.transfer(tx_nop, rx_data, 2);
411
412 // Process response
413 uint16_t raw = (static_cast<uint16_t>(rx_data[0]) << 8) | rx_data[1];
414 result = raw & 0x3FFF; // Mask out status flags
415 } else if (this->frame_format_ == FrameFormat::SPI_24) {
416 // ---- 24-bit frame with CRC ----
417 // Prepare read command with CRC
418 uint16_t crc_input = static_cast<uint16_t>((1 << 14) | (address & 0x3FFF));
419 uint8_t crc = ComputeCRC8(crc_input);
420 const uint8_t tx_cmd[3] = {
421 static_cast<uint8_t>(((address >> 8) & 0x3F) | 0x40), // bit6=1 for read
422 static_cast<uint8_t>(address & 0xFF), crc};
423 uint8_t rx_cmd[3];
424 spi_.transfer(tx_cmd, rx_cmd, 3);
425
426 // Send NOP to receive register data
427 uint16_t nop_addr = AS5047U_REG::NOP::ADDRESS;
428 uint16_t nop_crc_input = static_cast<uint16_t>((1 << 14) | (nop_addr & 0x3FFF));
429 uint8_t crc_nop = ComputeCRC8(nop_crc_input);
430 const uint8_t tx_nop[3] = {static_cast<uint8_t>(((nop_addr >> 8) & 0x3F) | 0x40),
431 static_cast<uint8_t>(nop_addr & 0xFF), crc_nop};
432 uint8_t rx_data_frame[3];
433 spi_.transfer(tx_nop, rx_data_frame, 3);
434
435 // Process response with CRC verification
436 uint16_t raw = (static_cast<uint16_t>(rx_data_frame[0]) << 8) | rx_data_frame[1];
437 uint8_t crc_device = rx_data_frame[2];
438 uint8_t crc_calc = ComputeCRC8(raw);
439 if (crc_device != crc_calc) {
440 // crc error, caller will read ERRFL
441 }
442 result = raw & 0x3FFF;
443 } else if (this->frame_format_ == FrameFormat::SPI_32) {
444 // ---- 32-bit frame with CRC and pad byte ----
445 // Prepare read command with CRC and pad
446 uint16_t crc_input = static_cast<uint16_t>((1 << 14) | (address & 0x3FFF));
447 uint8_t crc = ComputeCRC8(crc_input);
448 const uint8_t tx_cmd[4] = {
449 this->pad_byte_,
450 static_cast<uint8_t>(((address >> 8) & 0x3F) | 0x40), // bit6=1 for read
451 static_cast<uint8_t>(address & 0xFF), crc};
452 uint8_t rx_cmd[4];
453 spi_.transfer(tx_cmd, rx_cmd, 4);
454
455 // Send NOP to receive register data
456 uint16_t nop_addr = AS5047U_REG::NOP::ADDRESS;
457 uint16_t nop_crc_input = static_cast<uint16_t>((1 << 14) | (nop_addr & 0x3FFF));
458 uint8_t crc_nop = ComputeCRC8(nop_crc_input);
459 const uint8_t tx_nop[4] = {this->pad_byte_,
460 static_cast<uint8_t>(((nop_addr >> 8) & 0x3F) | 0x40),
461 static_cast<uint8_t>(nop_addr & 0xFF), crc_nop};
462 uint8_t rx_data_frame[4];
463 spi_.transfer(tx_nop, rx_data_frame, 4);
464
465 // Process response with CRC verification
466 uint16_t raw = (static_cast<uint16_t>(rx_data_frame[1]) << 8) | rx_data_frame[2];
467 uint8_t crc_device = rx_data_frame[3];
468 uint8_t crc_calc = ComputeCRC8(raw);
469 if (crc_device != crc_calc) {
470 // crc error, caller will read ERRFL
471 }
472 result = raw & 0x3FFF;
473 }
474 return result;
475}

486template <typename SpiType>
487bool AS5047U<SpiType>::writeRegister(uint16_t address, uint16_t value, uint8_t retries) const {
488 bool success = false;
489 uint16_t err_mask = static_cast<uint16_t>(AS5047U_Error::CrcError) |
490 static_cast<uint16_t>(AS5047U_Error::FramingError);
491 for (uint8_t attempt = 0; attempt <= retries; ++attempt) {
492 if (this->frame_format_ == FrameFormat::SPI_16) {
493 // ---- 16-bit write (two frames) ----
494 // First frame: send address
495 uint16_t cmd = static_cast<uint16_t>(address & 0x3FFF); // bit14=0 for write
496 uint8_t tx[2] = {static_cast<uint8_t>(cmd >> 8), static_cast<uint8_t>(cmd & 0xFF)};
497 uint8_t rx_dummy[2];
498 spi_.transfer(tx, rx_dummy, 2);
499
500 // Second frame: send data payload
501 uint16_t data_frame = value & 0x3FFF;
502 tx[0] = static_cast<uint8_t>(data_frame >> 8);
503 tx[1] = static_cast<uint8_t>(data_frame & 0xFF);
504 uint8_t rx_data_resp[2];
505 spi_.transfer(tx, rx_data_resp, 2);
506
507 // Check for errors from previous command
508 auto err = this->template ReadReg<AS5047U_REG::ERRFL>().value;
509 if (!(err & err_mask)) {
510 success = true;
511 break;
512 }
513 updateStickyErrors(err);
514 } else if (this->frame_format_ == FrameFormat::SPI_24) {
515 // ---- 24-bit write with CRC ----
516 // First frame: send address with CRC
517 uint16_t cmd_payload = static_cast<uint16_t>(address & 0x3FFF);
518 uint8_t cmd_crc = ComputeCRC8(cmd_payload);
519 const uint8_t tx_cmd[3] = {static_cast<uint8_t>((address >> 8) & 0x3F), // bit6=0 for write
520 static_cast<uint8_t>(address & 0xFF), cmd_crc};
521 uint8_t rx_cmd[3];
522 spi_.transfer(tx_cmd, rx_cmd, 3);
523
524 // Second frame: send data with CRC
525 uint16_t data_payload = value & 0x3FFF;
526 uint8_t data_crc = ComputeCRC8(data_payload);
527 const uint8_t tx_data[3] = {static_cast<uint8_t>((data_payload >> 8) & 0xFF),
528 static_cast<uint8_t>(data_payload & 0xFF), data_crc};
529 uint8_t rx_data[3];
530 spi_.transfer(tx_data, rx_data, 3);
531
532 // after write, check error flags
533 auto err = this->template ReadReg<AS5047U_REG::ERRFL>().value;
534 if (!(err & err_mask)) {
535 success = true;
536 break;
537 }
538 updateStickyErrors(err);
539 } else if (this->frame_format_ == FrameFormat::SPI_32) {
540 // ---- 32-bit write with CRC and pad byte ----
541 // First frame: send address with CRC and pad
542 uint16_t cmd_payload = static_cast<uint16_t>(address & 0x3FFF);
543 uint8_t cmd_crc = ComputeCRC8(cmd_payload);
544 const uint8_t tx_cmd[4] = {this->pad_byte_,
545 static_cast<uint8_t>((address >> 8) & 0x3F), // bit6=0 for write
546 static_cast<uint8_t>(address & 0xFF), cmd_crc};
547 uint8_t rx_cmd[4];
548 spi_.transfer(tx_cmd, rx_cmd, 4);
549
550 // Second frame: send data with CRC and pad
551 uint16_t data_payload = value & 0x3FFF;
552 uint8_t data_crc = ComputeCRC8(data_payload);
553 const uint8_t tx_data[4] = {this->pad_byte_, static_cast<uint8_t>((data_payload >> 8) & 0xFF),
554 static_cast<uint8_t>(data_payload & 0xFF), data_crc};
555 uint8_t rx_data[4];
556 spi_.transfer(tx_data, rx_data, 4);
557
558 // after write, check error flags
559 auto err = this->template ReadReg<AS5047U_REG::ERRFL>().value;
560 if (!(err & err_mask)) {
561 success = true;
562 break;
563 }
564 updateStickyErrors(err);
565 }
566 }
567 return success;
568}
