99 static constexpr uint8_t ComputeCRC8(uint16_t data16) {
100 uint8_t crc = 0xC4;
101 for (int i = 0; i < 16; ++i) {
102 bool bit = ((data16 >> (15 - i)) & 1) ^ ((crc >> 7) & 1);
103 crc = (crc << 1) ^ (bit ? 0x1D : 0x00);
104 }
105 return crc ^ 0xFF;
106 }
