143 static constexpr uint8_t computeCRC8(uint16_t data16) {
144 uint8_t crc = 0xC4;
145 for (int i = 0; i < 16; ++i) {
146 bool bit = ((data16 >> (15 - i)) & 1) ^ ((crc >> 7) & 1);
147 crc = (crc << 1) ^ (bit ? 0x1D : 0x00);
148 }
149 return crc ^ 0xFF;
150 }
