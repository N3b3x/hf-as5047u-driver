96 static constexpr uint8_t ComputeCRC8(uint16_t data16) {
97 uint8_t crc = 0xC4;
98 for (int i = 0; i < 16; ++i) {
99 bool bit = ((data16 >> (15 - i)) & 1) ^ ((crc >> 7) & 1);
100 crc = (crc << 1) ^ (bit ? 0x1D : 0x00);
101 }
102 return crc ^ 0xFF;
103 }
