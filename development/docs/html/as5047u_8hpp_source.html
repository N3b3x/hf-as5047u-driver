83 static constexpr uint8_t ComputeCRC8(uint16_t data16) {
84 uint8_t crc = 0xC4;
85 for (int i = 0; i < 16; ++i) {
86 bool bit = ((data16 >> (15 - i)) & 1) ^ ((crc >> 7) & 1);
87 crc = (crc << 1) ^ (bit ? 0x1D : 0x00);
88 }
89 return crc ^ 0xFF;
90 }
