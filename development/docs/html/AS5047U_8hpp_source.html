98 static constexpr uint8_t computeCRC8(uint16_t data16) {
99 uint8_t crc = 0xC4;
100 for (int i = 0; i < 16; ++i) {
101 bool bit = ((data16 >> (15 - i)) & 1) ^ ((crc >> 7) & 1);
102 crc = (crc << 1) ^ (bit ? 0x1D : 0x00);
103 }
104 return crc ^ 0xFF;
105 }
