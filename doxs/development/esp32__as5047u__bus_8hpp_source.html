166 bool initializeSPI() {
167 // Verify pin numbers are valid before configuring
168 if (config_.sclk_pin < 0 || config_.sclk_pin > 48) {
169 ESP_LOGE(TAG, "Invalid SCLK pin: %d", config_.sclk_pin);
170 return false;
171 }
172 if (config_.mosi_pin < 0 || config_.mosi_pin > 48) {
173 ESP_LOGE(TAG, "Invalid MOSI pin: %d", config_.mosi_pin);
174 return false;
175 }
176 if (config_.miso_pin < 0 || config_.miso_pin > 48) {
177 ESP_LOGE(TAG, "Invalid MISO pin: %d", config_.miso_pin);
178 return false;
179 }
180
181 spi_bus_config_t buscfg = {};
182 buscfg.mosi_io_num = config_.mosi_pin;
183 buscfg.miso_io_num = config_.miso_pin;
184 buscfg.sclk_io_num = config_.sclk_pin;
185 buscfg.quadwp_io_num = -1;
186 buscfg.quadhd_io_num = -1;
187 buscfg.max_transfer_sz = 64;
188 buscfg.flags = SPICOMMON_BUSFLAG_MASTER;
189
190 ESP_LOGI(TAG, "Initializing SPI bus: MISO=GPIO%d, MOSI=GPIO%d, SCLK=GPIO%d, Host=%d",
191 config_.miso_pin, config_.mosi_pin, config_.sclk_pin, config_.host);
192
193 esp_err_t ret = spi_bus_initialize(config_.host, &buscfg, SPI_DMA_CH_AUTO);
194 if (ret != ESP_OK) {
195 ESP_LOGE(TAG, "Failed to initialize SPI bus: %s", esp_err_to_name(ret));
196 ESP_LOGE(TAG, "Check: SCLK pin GPIO%d is not used by another peripheral", config_.sclk_pin);
197 return false;
198 }
199
200 ESP_LOGI(TAG, "SPI bus initialized successfully: MISO=GPIO%d, MOSI=GPIO%d, SCLK=GPIO%d",
201 config_.miso_pin, config_.mosi_pin, config_.sclk_pin);
202
203 // If you see CS/MOSI/MISO but no SCLK on GPIO%d:
204 // - Ensure SPI frequency > 0 (addSPIDevice checks this).
205 // - On ESP32-C6, GPIO12 is USB-JTAG; use another pin or disable USB-JTAG.
206 // - If SPI2 is used elsewhere, try SPI3_HOST in CreateEsp32As5047uBus().
207 // - Confirm no other driver (e.g. PSRAM, display) has claimed this GPIO.
208 ESP_LOGI(TAG, "SCLK pin GPIO%d is driven by SPI%d; freq=%lu Hz",
209 config_.sclk_pin, config_.host, (unsigned long)config_.frequency);
210
211 return true;
212 }

218 bool addSPIDevice() {
219 if (config_.frequency == 0) {
220 ESP_LOGE(TAG, "SPI frequency is 0 - SCLK will not run. Set SPIParams::FREQUENCY > 0 (e.g. 4000000)");
221 return false;
222 }
223 spi_device_interface_config_t devcfg = {};
224 devcfg.command_bits = 0;
225 devcfg.address_bits = 0;
226 devcfg.dummy_bits = 0;
227 devcfg.clock_speed_hz = config_.frequency;
228 devcfg.mode = config_.mode;
229 devcfg.duty_cycle_pos = 128; // 50% duty cycle
230 devcfg.spics_io_num = config_.cs_pin;
231 devcfg.queue_size = config_.queue_size;
232 devcfg.cs_ena_pretrans = config_.cs_ena_pretrans;
233 devcfg.cs_ena_posttrans = config_.cs_ena_posttrans;
234 devcfg.flags = 0; // No special flags
235 devcfg.input_delay_ns = 0;
236 devcfg.pre_cb = nullptr;
237 devcfg.post_cb = nullptr;
238
239 ESP_LOGI(TAG, "Adding SPI device: CS=GPIO%d, Freq=%lu Hz, Mode=%d, SCLK=GPIO%d",
240 config_.cs_pin, config_.frequency, config_.mode, config_.sclk_pin);
241
242 esp_err_t ret = spi_bus_add_device(config_.host, &devcfg, &spi_device_);
243 if (ret != ESP_OK) {
244 ESP_LOGE(TAG, "Failed to add SPI device: %s", esp_err_to_name(ret));
245 ESP_LOGE(TAG, "Check: CS pin GPIO%d and SCLK pin GPIO%d are not used elsewhere",
246 config_.cs_pin, config_.sclk_pin);
247 return false;
248 }
249
250 ESP_LOGI(TAG, "SPI device added successfully: CS=GPIO%d, Freq=%lu Hz, Mode=%d",
251 config_.cs_pin, config_.frequency, config_.mode);
252
253 return true;
254 }

265inline auto CreateEsp32As5047uBus() noexcept -> std::unique_ptr<Esp32As5047uBus> {
266 using namespace AS5047U_TestConfig;

268 Esp32As5047uBus::SPIConfig config;

272 config.miso_pin = static_cast<gpio_num_t>(SPIPins::MISO);
273 config.mosi_pin = static_cast<gpio_num_t>(SPIPins::MOSI);
274 config.sclk_pin = static_cast<gpio_num_t>(SPIPins::SCLK);
275 config.cs_pin = static_cast<gpio_num_t>(SPIPins::CS);
276
277 // SPI parameters from esp32_as5047u_test_config.hpp
278 config.frequency = SPIParams::FREQUENCY;
279 config.mode = SPIParams::MODE;
280 config.queue_size = SPIParams::QUEUE_SIZE;
281 config.cs_ena_pretrans = SPIParams::CS_ENA_PRETRANS;
282 config.cs_ena_posttrans = SPIParams::CS_ENA_POSTTRANS;
283
284 return std::make_unique<Esp32As5047uBus>(config);
